# Angular Web UX Architecture

## Overview

The Inception Web UX provides a browser-based interface mirroring the CLI/TUI functionality, enabling:

- **File upload and ingestion** without VPS command-line access
- **Knowledge graph exploration** with visual graph rendering
- **CLI passthrough** via WebSocket terminal emulation
- **OAuth management** for provider authentication

---

## Technology Stack

| Layer | Technology | Purpose |
|-------|------------|---------|
| **Framework** | Angular 17+ (Standalone) | Component architecture |
| **UI Library** | Angular Material 17+ | Core components |
| **Design System** | Material Web 3 | Modern M3 tokens |
| **State** | Angular Signals | Reactive state management |
| **Terminal** | xterm.js | Terminal emulation |
| **Graph** | Cytoscape.js / D3.js | Knowledge graph visualization |
| **HTTP** | Angular HttpClient | API communication |
| **WebSocket** | RxJS WebSocket | CLI passthrough |

## API Service Layer

```typescript
// services/knowledge.service.ts
@Injectable({ providedIn: 'root' })
export class KnowledgeService {
  private readonly http = inject(HttpClient);
  private readonly baseUrl = '/api/v1';
  
  // Signals for reactive state
  readonly entities = signal<Entity[]>([]);
  readonly claims = signal<Claim[]>([]);
  readonly loading = signal<boolean>(false);
  
  async loadEntities(filters?: EntityFilters): Promise<void> {
    this.loading.set(true);
    try {
      const result = await this.http.get<Entity[]>(
        `${this.baseUrl}/entities`,
        { params: filters as any }
      ).toPromise();
      this.entities.set(result);
    } finally {
      this.loading.set(false);
    }
  }
  
  async search(query: string): Promise<SearchResult[]> {
    return this.http.get<SearchResult[]>(
      `${this.baseUrl}/search`,
      { params: { q: query } }
    ).toPromise();
  }
}
```

## WebSocket Terminal Bridge

```typescript
// services/terminal.service.ts
@Injectable({ providedIn: 'root' })
export class TerminalService {
  private socket: WebSocket | null = null;
  
  readonly output = signal<string[]>([]);
  readonly connected = signal<boolean>(false);
  
  connect(wsUrl: string): void {
    this.socket = new WebSocket(wsUrl);
    
    this.socket.onopen = () => this.connected.set(true);
    this.socket.onclose = () => this.connected.set(false);
    
    this.socket.onmessage = (event) => {
      this.output.update(lines => [...lines, event.data]);
    };
  }
  
  sendCommand(command: string): void {
    this.socket?.send(JSON.stringify({ type: 'command', data: command }));
  }
  
  uploadFile(file: File): Observable<UploadProgress> {
    // Stream file via WebSocket with progress
  }
}
```

```nginx
server {
    listen 443 ssl;
    server_name inception.example.com;
    
    location / {
        root /var/www/inception;
        try_files $uri $uri/ /index.html;

    location /api {

    location /ws {
        proxy_pass http://localhost:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}
```
